189 iterator insert(iterator position, iterator _begin, iterator _end) override {
190 auto length = _end - _begin;
191 if (!_own_iterator(position) || length + size() > N)
192 return end();
193
194 if (_own_iterator(_begin) && _own_iterator(_end))
195 return _insert_overlap(position, _begin, _end);
196
197 // displace elements
198 for (auto iter = end() + length - 1; iter > position; --iter) {
199 *iter = utility::move(*(iter - length));
