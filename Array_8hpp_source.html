231 iterator insert(iterator position, iterator _begin, iterator _end) override {
232 if (!_validate_position(position) || _begin > _end || _end - _begin > end() - position)
233 return end();
234
235 // different logic if inserting into itself
236 if (_own_iterator(_begin) && _own_iterator(_end))
237 return _insert_overlap(position, _begin, _end);
238
239 auto remaining_till_end = end() - position; // remaining elements till end from position
240 auto new_elems = _end - _begin; // number of new elements to insert
241 auto to_displace = remaining_till_end - new_elems; // number of element that will need to be displaced
242
243 // displace elements to make room
244 auto iter = end() - 1;
245 while (to_displace--) {
246 *iter = utility::move(*(iter - new_elems));
247 --iter;
248 }
249
250 // copy new elements
251 iter = position;
252 while (_begin != _end) {
253 *iter = *_begin;
254 ++_begin;
255 ++iter;
256 }
257
258 return position;
259 }

265 iterator _insert_overlap(iterator position, iterator _begin, iterator _end) {
266 auto new_elems = _end - _begin;
267 auto distance = _begin - position;
268
269 auto src_iter = _begin;
270 auto dst_iter = src_iter - distance;
271
272 // displace elements
273 while (new_elems--) {
274 utility::swap(*src_iter, *dst_iter);
275 ++src_iter;
276 ++dst_iter;
277 }
278
279 return position;
280 }
